    # 数的乘积

     //首先看看迭代乘积
function factorial(number){
    if(number <0) return undefined;
    let total = 1;
    for (let n = number; n>1; n--) {
      total*= n;
    }
    return total
}

    递归版本

    function factorialx(n){
   if(n===1 || n === 0){
     return 1
   }
   return n * factorialx(n-1)
}

console.log('递归：',factorialx(5));

    # 斐波那契

    数列规律是这样的：0 1 1 2 3 5 8 13 21

    function fibonacci(n){
  if(n<1) return 0  ;
  if(n <=2) return 1;
  return fibonacci(n-1) + fibonacci(n-2)
}

console.log(fibonacci(5));

    记忆化斐波那契

    function betterFibonacci(n){
    const memo = [0,1]   //记录缓存
    const fibonacci = (n)=>{
       if(memo[n] != null) return memo[n]; //如果有缓存 则返回它
        return memo[n] = fibonacci(n-1,memo) + fibonacci(n-2,memo)
    }
    return fibonacci
}

console.log(betterFibonacci(5));

    # js调用栈

    递归不会被无限制循环，当达到浏览器上限值的时候，就会报错终止，所谓的栈溢出错误 每个浏览器都有自己的限制比如：谷歌 15662个 火狐 188641个 Edge 17654个

    调用栈是如何工作呢？ 每当一个函数被一个算法调用时，该函数就会进入调用栈的顶部,每当使用递归的时候，每一个函数都堆叠到顶部，这是因为每一个调用都依赖前一个调用的结果
