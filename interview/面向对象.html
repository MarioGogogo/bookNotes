    # 借助构造函数来实现继承

    function Parent(){
  this.name="父亲"
}
function Son(){
  Parent.call(this)
  this.type="child_son";
}
Parent.prototype.sany=function(){this.say="hello"}

    将父级的构造函数的this指向，到子构造函数的实例上去，导致父级上的属性和方法，在子类中也有， （缺点：如果父级有prototype上的方法，子类的this还是指向不到的）

    # 借助原型链来继承

    function Parent(){
   this.name="parent"
}
function Son(){
  this.type="man"
}
//parent的实例 赋值给 son的原型对象
Son.prototype = new Parent();
new Son()._proto_ === new Parent();//true

    缺点：如果创建多个子类的实例，如果更新第一个子类的实例中的数组，那么其他子类的数组也会改变；因为引用的同一个原型对象是共享的

    # 组合继承

    function Parent3(){
  this.name='p3';
  this.play=[1,2,3]
}
function Child3(){
  Parent3.call(this)
  this.type="child3"
}
Child3.prototype = new Parent3()
var s3 = new Child3()
var s4 = new Child3()
s3.play.push(4) //[1,2,3,4]
s4 //[1,2,3]

    优化1 既然想继承父类的原型对象的属性和方法都给我就行了

    Child3.prototype = Parent3.prototype

    缺点：s3 s4的 instanceof constructor 都是指向 Parent4 无法区分实例由父类创建还是子类创建

    优化2

    Child3.prototype = Object.create(Parent3.prototype)
Child3.prototype.constructio = Child3;

    var s5 = new Child3(); s5 instanceof Child3 //true
